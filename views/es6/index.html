Promises

MDN Promise spec

The Promise object is used for deferred and asynchronous computations. 

          
            var p = new Promise(
              function(resolve, reject){
                alert('first function begins and takes 2 second to complete');
                setTimeout(resolve, 2000);
              }
            );

            p.then(function(){
              alert('next function fires only after original function finishes');
            });

Block-scoped Variables

MDN Let statement spec

The let statement declares a block scope local variable, optionally initializing it to a value. 

          
            function order(x, y) {
              if (x > y) {
                var tmp = x;
                x = y;
                y = tmp;
              }
              alert(tmp);
              return [x,y];
            }
          

          
            function orderNew(x, y) {
              if (x > y) {
                let tmp = x;
                x = y;
                y = tmp;
              }
              alert(tmp);
              return [x,y];
            }

Sets

MDN Set object spec

The Set object lets you store unique values of any type, whether primitive values or object references. 

          
            var foo = new Set();

            foo.add(1);
            foo.add(5);
            foo.add("some text");
            foo.has(1);
            foo.has(3);
            foo.has(5);
            foo.has(Math.sqrt(25));
            foo.has("Some Text".toLowerCase());
            foo.size;
            foo.delete(5);
            foo.has(5);
            foo.size;
            for (let item of foo) console.log(item);
            foo.forEach(function(value) {
              console.log(value);
            });

Template strings

MDN Template strings spec

Template strings are string literals allowing embedded expressions. You can use multi-line strings and string interpolation features with them. 

          
          // coming in Firefox 34 (current is 32)
          var a = 5;
          var b = 10;
          alert(`Fifteen is ${a + b} and not ${2 * a + b}.`);
          // "Fifteen is 15 and not 20."
